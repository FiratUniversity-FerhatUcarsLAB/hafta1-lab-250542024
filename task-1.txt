Lütfen seçtiğiniz algoritmaya ait çözümü ve diğer isterleri aşağıya ekleyiniz:
ALGORİTMA Mükemmel_Omlet_Yap

BAŞLA

    // 1. Malzemeleri hazırlama
    OLUŞTUR kase
    YUMURTALARI_KIR(kase, Yumurta_sayisi)
    EKLE(kase, Tuz_miktari)
    EKLE(kase, Süt_miktari)     // yumuşaklık için isteğe bağlı
    Karışım_homojen_mi ← FALSE

    // 2. Karışımı çırpma
    WHILE Karışım_homojen_mi = FALSE DO
        ÇIRP(kase)
        EĞER KARIŞIM_HOMOJEN(kase) İSE
            Karışım_homojen_mi ← TRUE
        SON
    ENDWHILE

    // 3. Tavayı hazırlama
    ISIT(tava, Tava_sicakligi)
    EKLE(tava, Tereyagi_miktari)
    BEKLE(30 saniye)
    EĞER YAĞ_YANIYOR İSE
        SICAKLIK_AZALT(tava)
    SON

    // 4. Karışımı tavaya dökme
    DÖK(tava, kase_karışımı)
    BEKLE(10 saniye)

    // 5. Omleti şekillendirme ve pişirme
    TEKRARLA
        SPATULA_ILE(Kenarları_İçe_İt)
        EĞİM_VER(tava)
        AKIŞKAN_KISIM_BOŞLUĞA_AKSIN
    UNTIL ALT_KISIM_ALTIN_RENGİ_ALDI

    // 6. İç malzeme ekleme (isteğe bağlı)
    EĞER Ic_malzeme ≠ BOŞ İSE
        EKLE(omletin_yarısı, Ic_malzeme)
        KATLA(omlet, İkiye)
    SON

    // 7. Son pişirme kontrolü
    EĞER ÜST_KISIM_HAFİF_NEMLİ İSE
        BEKLE(20–30 saniye)
    SON

    // 8. Servis
    AL(tava, omlet)
    SÜSLE(omlet, karabiber + maydanoz)
    SERVİS_ET(omlet)

    // 9. Son kontrol
    EĞER ALT_YANIK_DEĞİL VE DOKU_YUMUŞAK İSE
        Omlet_durumu ← “Mükemmel omlet hazır 🍳”
    DEĞİLSE
        Omlet_durumu ← “Isı veya pişirme süresi ayarlanmalı.”
    SON

    YAZ(Omlet_durumu)

BİTİR

SON ALGORİTMA
